START
PRINT "*** Welcome to REAL CHAMPION SPORT ACADEMY! ***"
PRINT "Who are you?"
PRINT "1. Admin"
PRINT "2. All Students (Registered / Not-Registered)"
PRINT "3. Registered Student"

READ user
TRY
  user = int(user)
  IF user == 1 THEN
    IF call: ad.admin_login() THEN
      DOWHILE 1
        PRINT "*** Admin Menu ***"
        PRINT "1. Add Record"
        PRINT "2. Display All Records"
        PRINT "3. Search Specific Record"
        PRINT "4. Sort and Display Record"
        PRINT "5. Modify Record"
        PRINT "6. Exit"
        READ choice
        TRY
          choice = int(choice)
          IF choice == 1 THEN
            call: admin_add()
          ELSEIF choice == 2 THEN
            call: admin_display()
          ELSEIF choice == 3 THEN
            call: admin_search()
          ELSEIF choice == 4 THEN
            call: admin_sort()
          ELSEIF choice == 5 THEN
            call: admin_modify()
          ELSEIF choice == 6 THEN
            PRINT "*** Log Out ***"
            BREAK
          ELSE
            PRINT "Please enter 1 ~ 6"
            CONTINUE
          ENDIF
        EXCEPTION
          PRINT str(choice), "is not a number"
          CONTINUE
        ENDEXCEPTION
      ENDDO
    ENDIF
  ELSEIF user == 2 THEN 
    DOWHILE
      PRINT "*** Student Menu ***"
      PRINT "1. View details of..."
      PRINT "2. If new student Register to Access other  Details"
      PRINT "3. Exit"
      READ choice
      TRY
        choice = int(choice)
        IF choice == 1 THEN
          call: view_details()
        ELSEIF choice == 2 THEN
          call: student_signup()
        ELSEIF choice == 3 THEN
          PRINT "*** Log Out ***"
        ELSE
          PRINT "Please enter 1 ~ 3"
          CONTINUE
        ENDIF
      EXCEPTION
        PRINT str(choice), "is not a number"
        CONTINUE
      ENDTRYEXCEPTION
    ENDDO
  ELSEIF user == 3 THEN
    SET success, student_id to call: student_login()
    IF success == true THEN
      DOWHILE 1
        PRINT "*** Resistered Student Menu ***"
        PRINT "1. View Detail of"
        PRINT "2. Modify Self Record"
        PRINT "3. Provide feedback and Star to Coach"
        PRINT "4. Exit"
        READ choice
        TRY
          choice = int(choice)
          IF choice == 1 THEN
            call: view_details(student_id)
          ELSEIF choice == 2 THEN
            SET student_id to call: modify_selfrecord(student_id)
          ELSEIF choice == 3 THEN
            call: feedback_star(student_id)
          ELSEIF choice == 4 THEN 
            PRINT "*** Log Out ***"
            BREAK
          ELSE
            PRINT "Please enter 1 ~ 4"
            CONTINUE
          ENDIF
        EXCEPTION
          PRINT str(choice), "is not a number"
          CONTINUE
        ENDTRYEXCEPTION
      ENDDO
    ENDIF
  ELSE
    PRINT "Please enter 1 ~ 3"
  ENDIF
EXCEPTION
  PRINT str(user), "is not a number"
ENDEXCEPTION
PRINT "★★★ See You ★★★"
END
